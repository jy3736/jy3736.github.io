1. 變數與基本型別

A. 宣告、初始化、輸出

C 變數必須先宣告再使用，且型別會決定記憶體大小與可表示範圍。

#include <stdio.h>

int main(void) {
    int age = 20;
    double height = 172.5;
    char grade = 'A';

    printf("age=%d, height=%.1f, grade=%c\n", age, height, grade);
    return 0;
}
#include <stdio.h>

int main(void) {
    int a = 3, b = 4;
    int sum = a + b;
    printf("sum=%d\n", sum);
    return 0;
}
#include <stdio.h>

int main(void) {
    const double PI = 3.14159;
    double r = 2.0;
    printf("area=%.2f\n", PI * r * r);
    return 0;
}

注意事項：

- char 用 %c，int 用 %d，double 用 %f。

- const 常數不可被重新指定。

B. 型別轉換與 sizeof

#include <stdio.h>

int main(void) {
    int x = 7, y = 2;
    printf("int division: %d\n", x / y);       // 3
    printf("float division: %.2f\n", (double)x / y); // 3.50
    return 0;
}
#include <stdio.h>

int main(void) {
    char ch = 'A';
    int code = ch;
    printf("%c -> %d\n", ch, code);
    return 0;
}
#include <stdio.h>

int main(void) {
    printf("sizeof(int)=%zu\n", sizeof(int));
    printf("sizeof(double)=%zu\n", sizeof(double));
    return 0;
}

注意事項：

- int / int 會做整數除法。

- sizeof 回傳型別是 size_t，通常以 %zu 輸出。

2. 運算式與運算子

A. 算術、關係、邏輯運算

#include <stdio.h>

int main(void) {
    int a = 10, b = 3;
    printf("+ - * / %% -> %d %d %d %d %d\n", a + b, a - b, a * b, a / b, a % b);
    return 0;
}
#include <stdio.h>

int main(void) {
    int score = 78;
    printf("score >= 60 ? %d\n", score >= 60); // 1(true)
    printf("score < 60 ? %d\n", score < 60);   // 0(false)
    return 0;
}
#include <stdio.h>

int main(void) {
    int age = 19, has_id = 1;
    if (age >= 18 && has_id) {
        printf("entry allowed\n");
    }
    return 0;
}

注意事項：

- C 沒有布林字面值時可用 0/1 表示真假（或使用 <stdbool.h>）。

- % 只適用於整數。

B. 優先順序與遞增運算

#include <stdio.h>

int main(void) {
    int result = 2 + 3 * 4;
    printf("2 + 3 * 4 = %d\n", result); // 14
    return 0;
}
#include <stdio.h>

int main(void) {
    int result = (2 + 3) * 4;
    printf("(2 + 3) * 4 = %d\n", result); // 20
    return 0;
}
#include <stdio.h>

int main(void) {
    int i = 5;
    printf("before: i=%d\n", i);
    printf("postfix value=%d\n", i++);
    printf("after postfix: i=%d\n", i);
    i = 5;
    printf("prefix value=%d\n", ++i);
    printf("after prefix: i=%d\n", i);
    return 0;
}

注意事項：

- 先乘除後加減；不確定時加括號。

- i++ 先用後加，++i 先加後用。

3. 判斷流程

A. if / else if / else

#include <stdio.h>

int main(void) {
    int score = 85;
    if (score >= 90) {
        printf("A\n");
    } else if (score >= 80) {
        printf("B\n");
    } else if (score >= 70) {
        printf("C\n");
    } else {
        printf("D\n");
    }
    return 0;
}
#include <stdio.h>

int main(void) {
    int n = -3;
    if (n > 0) {
        printf("positive\n");
    } else if (n < 0) {
        printf("negative\n");
    } else {
        printf("zero\n");
    }
    return 0;
}
#include <stdio.h>

int main(void) {
    int year = 2024;
    if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
        printf("leap year\n");
    } else {
        printf("not leap year\n");
    }
    return 0;
}

注意事項：

- 複合條件建議加括號提升可讀性。

- = 是指定，== 才是比較。

B. switch 使用情境

#include <stdio.h>

int main(void) {
    int menu = 2;
    switch (menu) {
        case 1: printf("Add\n"); break;
        case 2: printf("Edit\n"); break;
        case 3: printf("Delete\n"); break;
        default: printf("Unknown\n");
    }
    return 0;
}
#include <stdio.h>

int main(void) {
    char grade = 'B';
    switch (grade) {
        case 'A':
        case 'B':
            printf("Pass with good result\n");
            break;
        case 'C':
            printf("Pass\n");
            break;
        default:
            printf("Need improvement\n");
    }
    return 0;
}
#include <stdio.h>

int main(void) {
    int month = 2;
    int days;
    switch (month) {
        case 2: days = 28; break;
        case 4: case 6: case 9: case 11: days = 30; break;
        default: days = 31;
    }
    printf("days=%d\n", days);
    return 0;
}

注意事項：

- switch 條件需是整數型別（含 char）。

- 忘記 break 會發生 fall-through。

4. 迴圈

A. for / while / do...while

#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 5; i++) {
        printf("%d ", i);
    }
    printf("\n");
    return 0;
}
#include <stdio.h>

int main(void) {
    int n = 5;
    int sum = 0;
    while (n > 0) {
        sum += n;
        n--;
    }
    printf("sum=%d\n", sum);
    return 0;
}
#include <stdio.h>

int main(void) {
    int x = 3;
    do {
        printf("x=%d\n", x);
        x--;
    } while (x > 0);
    return 0;
}

注意事項：

- for 適合已知次數，while 適合條件驅動。

- do...while 至少執行一次。

B. break、continue 與巢狀迴圈

#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 10; i++) {
        if (i == 6) break;
        printf("%d ", i);
    }
    printf("\n");
    return 0;
}
#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 10; i++) {
        if (i % 2 == 0) continue;
        printf("%d ", i);
    }
    printf("\n");
    return 0;
}
#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 3; i++) {
        for (int j = 1; j <= 3; j++) {
            printf("(%d,%d) ", i, j);
        }
        printf("\n");
    }
    return 0;
}

注意事項：

- break 只會跳出最近一層迴圈。

- 巢狀迴圈先外層再內層，時間複雜度常為乘法。

5. 函式與參數傳遞

A. 函式宣告、定義、呼叫

#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int main(void) {
    printf("%d\n", add(3, 5));
    return 0;
}
#include <stdio.h>

double circle_area(double r) {
    return 3.14159 * r * r;
}

int main(void) {
    printf("%.2f\n", circle_area(2.0));
    return 0;
}
#include <stdio.h>

void print_line(char c, int n) {
    for (int i = 0; i < n; i++) printf("%c", c);
    printf("\n");
}

int main(void) {
    print_line('-', 10);
    return 0;
}

注意事項：

- 先宣告（prototype）可讓函式放在 main 後面。

- void 回傳表示不回傳值。

B. 傳值與透過指標回傳結果

#include <stdio.h>

void increase(int x) {
    x++;
}

int main(void) {
    int a = 10;
    increase(a);
    printf("a=%d\n", a); // 仍是 10
    return 0;
}
#include <stdio.h>

void increase_by_pointer(int *x) {
    (*x)++;
}

int main(void) {
    int a = 10;
    increase_by_pointer(&a);
    printf("a=%d\n", a); // 11
    return 0;
}
#include <stdio.h>

void min_max(int a, int b, int *mn, int *mx) {
    if (a < b) {
        *mn = a; *mx = b;
    } else {
        *mn = b; *mx = a;
    }
}

int main(void) {
    int mn, mx;
    min_max(8, 3, &mn, &mx);
    printf("min=%d, max=%d\n", mn, mx);
    return 0;
}

注意事項：

- C 預設是 pass-by-value。

- 要修改呼叫端資料需傳入位址（pointer）。

6. 一維與二維陣列

A. 一維陣列

#include <stdio.h>

int main(void) {
    int scores[5] = {90, 75, 88, 100, 67};
    printf("first=%d, last=%d\n", scores[0], scores[4]);
    return 0;
}
#include <stdio.h>

int main(void) {
    int arr[] = {2, 4, 6, 8};
    int n = sizeof(arr) / sizeof(arr[0]);
    int sum = 0;
    for (int i = 0; i < n; i++) sum += arr[i];
    printf("sum=%d\n", sum);
    return 0;
}
#include <stdio.h>

int main(void) {
    int arr[] = {5, 1, 9, 3};
    int n = sizeof(arr) / sizeof(arr[0]);
    int mx = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > mx) mx = arr[i];
    }
    printf("max=%d\n", mx);
    return 0;
}

注意事項：

- 索引範圍是 0 到 n-1。

- 在同一個 scope 內可用 sizeof 計算元素數量。

B. 二維陣列

#include <stdio.h>

int main(void) {
    int mat[2][3] = {
        {1, 2, 3},
        {4, 5, 6}
    };
    printf("mat[1][2]=%d\n", mat[1][2]);
    return 0;
}
#include <stdio.h>

int main(void) {
    int mat[2][3] = {{1, 2, 3}, {4, 5, 6}};
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%d ", mat[i][j]);
        }
        printf("\n");
    }
    return 0;
}
#include <stdio.h>

int main(void) {
    int mat[2][3] = {{1, 2, 3}, {4, 5, 6}};
    int row_sum[2] = {0};
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            row_sum[i] += mat[i][j];
        }
    }
    printf("row sums: %d %d\n", row_sum[0], row_sum[1]);
    return 0;
}

注意事項：

- int a[R][C] 的欄數 C 在函式參數中通常必須已知。

- 走訪二維陣列常用雙層迴圈。

7. 簡單指標

A. 取址與解參照

#include <stdio.h>

int main(void) {
    int x = 42;
    int *p = &x;
    printf("x=%d, *p=%d\n", x, *p);
    return 0;
}
#include <stdio.h>

int main(void) {
    int x = 10;
    int *p = &x;
    *p = 99;
    printf("x=%d\n", x);
    return 0;
}
#include <stdio.h>

int main(void) {
    int x = 7;
    int *p = &x;
    printf("address=%p\n", (void*)p);
    return 0;
}

注意事項：

- & 取位址，* 解參照。

- 輸出指標位址時用 %p，並轉成 (void*)。

B. 指標與陣列關係

#include <stdio.h>

int main(void) {
    int arr[] = {10, 20, 30};
    int *p = arr;
    printf("%d %d %d\n", p[0], p[1], p[2]);
    return 0;
}
#include <stdio.h>

int main(void) {
    int arr[] = {5, 6, 7};
    int *p = arr;
    printf("first=%d\n", *p);
    p++;
    printf("second=%d\n", *p);
    return 0;
}
#include <stdio.h>

void fill_zero(int *arr, int n) {
    for (int i = 0; i < n; i++) arr[i] = 0;
}

int main(void) {
    int data[4] = {1, 2, 3, 4};
    fill_zero(data, 4);
    for (int i = 0; i < 4; i++) printf("%d ", data[i]);
    printf("\n");
    return 0;
}

注意事項：

- 陣列名在多數運算式會退化為指向首元素的指標。

- 指標位移是以元素大小為單位前進。

8. 綜合實作：成績統計系統

A. 實作目標

1. 讀入 5 位學生的 3 科成績（2D 陣列）。

2. 使用函式計算每位學生總分（1D 陣列）。

3. 用判斷分類及格與不及格。

4. 使用指標函式找出最高總分。

B. 完整程式碼範例

#include <stdio.h>

#define STUDENTS 5
#define SUBJECTS 3

void calc_totals(int scores[STUDENTS][SUBJECTS], int totals[STUDENTS]) {
    for (int i = 0; i < STUDENTS; i++) {
        totals[i] = 0;
        for (int j = 0; j < SUBJECTS; j++) {
            totals[i] += scores[i][j];
        }
    }
}

void find_max_total(int totals[STUDENTS], int *max_idx, int *max_val) {
    *max_idx = 0;
    *max_val = totals[0];
    for (int i = 1; i < STUDENTS; i++) {
        if (totals[i] > *max_val) {
            *max_val = totals[i];
            *max_idx = i;
        }
    }
}

int main(void) {
    int scores[STUDENTS][SUBJECTS] = {
        {80, 75, 90},
        {60, 58, 70},
        {95, 88, 92},
        {40, 65, 55},
        {78, 82, 79}
    };

    int totals[STUDENTS];
    calc_totals(scores, totals);

    printf("ID  Scores(3)        Total  Result\n");
    printf("-----------------------------------\n");
    for (int i = 0; i < STUDENTS; i++) {
        printf("%d   %3d %3d %3d    %3d    %s\n",
               i,
               scores[i][0], scores[i][1], scores[i][2],
               totals[i],
               (totals[i] >= 180) ? "PASS" : "FAIL");
    }

    int max_idx, max_val;
    find_max_total(totals, &max_idx, &max_val);
    printf("\nTop student: ID=%d, total=%d\n", max_idx, max_val);

    return 0;
}

C. 程式碼亮點解析

1. 二維陣列 + 一維陣列整合

for (int i = 0; i < STUDENTS; i++) {
    totals[i] = 0;
    for (int j = 0; j < SUBJECTS; j++) {
        totals[i] += scores[i][j];
    }
}

使用巢狀迴圈把每一列（每位學生）彙總為一個總分。

2. 用指標輸出多個結果

void find_max_total(int totals[STUDENTS], int *max_idx, int *max_val)

透過 int *max_idx 和 int *max_val 一次回傳兩個資訊。

3. 三元運算子做快速分類

(totals[i] >= 180) ? "PASS" : "FAIL"

在輸出時直接完成判斷，簡潔但仍可讀。
