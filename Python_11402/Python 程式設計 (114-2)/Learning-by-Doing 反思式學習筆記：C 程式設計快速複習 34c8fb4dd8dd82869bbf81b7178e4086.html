1. 變數與基本型別

概念摘要

C 是靜態型別語言，變數需先宣告。型別會影響計算結果、記憶體大小與輸出格式。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int main(void) {
    int age = 20;
    double height = 172.5;
    char grade = 'A';
    printf("%d %.1f %c\n", age, height, grade);
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

int main(void) {
    int x = 7, y = 2;
    printf("int div=%d\n", x / y);
    printf("float div=%.2f\n", (double)x / y);
    return 0;
}

3. 小任務練習

#include <stdio.h>

int main(void) {
    const double PI = 3.14159;
    double r = 5.0;
    // 練習：輸出圓面積到小數點後 2 位
    printf("area=%.2f\n", PI * r * r);
    return 0;
}

反思提問

- 為什麼 7 / 2 與 (double)7 / 2 的結果不同？

- 我是否能正確對應 %d、%f、%c？

- 哪些值應該使用 const？

小型練習

- 新增 long 或 float 變數並用正確格式輸出。

- 使用 sizeof 印出 int、double 大小。

2. 運算式與運算子

概念摘要

運算式由運算元與運算子組成，需特別注意優先順序與遞增運算行為。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int main(void) {
    int a = 10, b = 3;
    printf("+ - * / %% = %d %d %d %d %d\n", a+b, a-b, a*b, a/b, a%b);
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

int main(void) {
    printf("2 + 3 * 4 = %d\n", 2 + 3 * 4);
    printf("(2 + 3) * 4 = %d\n", (2 + 3) * 4);
    return 0;
}

3. 小任務練習

#include <stdio.h>

int main(void) {
    int i = 5;
    printf("before=%d\n", i);
    printf("postfix=%d\n", i++);
    printf("after postfix=%d\n", i);
    return 0;
}

反思提問

- 我何時應該主動加括號避免誤讀？

- i++ 與 ++i 在表達式中的差異是什麼？

- % 何時不能使用？

小型練習

- 計算分鐘轉秒、公里轉公尺。

- 自行設計一個含 && 與 || 的條件表達式。

3. 判斷流程（if / switch）

概念摘要

判斷流程讓程式能依資料做不同路徑。if 適合範圍條件，switch 適合離散選項。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int main(void) {
    int score = 85;
    if (score >= 60) printf("PASS\n");
    else printf("FAIL\n");
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

int main(void) {
    int score = 85;
    if (score >= 90) printf("A\n");
    else if (score >= 80) printf("B\n");
    else printf("C or below\n");
    return 0;
}

3. 小任務練習

#include <stdio.h>

int main(void) {
    int menu = 2;
    switch (menu) {
        case 1: printf("View\n"); break;
        case 2: printf("Edit\n"); break;
        default: printf("Unknown\n");
    }
    return 0;
}

反思提問

- 我是否曾把 == 寫成 =？如何避免？

- 哪種情境該用 switch，哪種該用 if？

- break 少寫會發生什麼？

小型練習

- 完成閏年判斷。

- 用 switch 寫月份天數查詢（先不考慮閏年）。

4. 迴圈（for / while / do...while）

概念摘要

迴圈用於重複執行。重點在正確設定起始值、終止條件與更新步驟。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 5; i++) printf("%d ", i);
    printf("\n");
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

int main(void) {
    int n = 5, sum = 0;
    while (n > 0) {
        sum += n;
        n--;
    }
    printf("sum=%d\n", sum);
    return 0;
}

3. 小任務練習

#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 10; i++) {
        if (i % 2 == 0) continue;
        if (i > 7) break;
        printf("%d ", i);
    }
    printf("\n");
    return 0;
}

反思提問

- 我的迴圈條件是否會造成無限迴圈？

- break 與 continue 對流程影響差在哪？

- 巢狀迴圈的執行次數如何估算？

小型練習

- 印出 1 到 9 的乘法表。

- 計算某數字各位數總和。

5. 函式與參數傳遞

概念摘要

函式能分解問題並提升重用性。C 預設傳值，若要修改外部資料需傳指標。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int main(void) {
    printf("%d\n", add(3, 4));
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

void increase(int x) {
    x++;
}

int main(void) {
    int a = 10;
    increase(a);
    printf("a=%d\n", a); // 不變
    return 0;
}

3. 小任務練習

#include <stdio.h>

void increase_by_ptr(int *x) {
    (*x)++;
}

int main(void) {
    int a = 10;
    increase_by_ptr(&a);
    printf("a=%d\n", a); // 變成 11
    return 0;
}

反思提問

- 為什麼 increase(a) 不會改到 a？

- 我是否能解釋 int *x 與 *x 的角色差異？

- 哪些邏輯適合抽成函式？

小型練習

- 寫一個 min_max 函式，透過指標回傳最小與最大值。

- 寫一個 is_even 函式回傳是否偶數。

6. 一維與二維陣列

概念摘要

陣列適合儲存同型別多筆資料；二維陣列可表示表格，常搭配巢狀迴圈。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int main(void) {
    int arr[] = {2, 4, 6, 8};
    int n = sizeof(arr) / sizeof(arr[0]);
    int sum = 0;
    for (int i = 0; i < n; i++) sum += arr[i];
    printf("sum=%d\n", sum);
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

int main(void) {
    int mat[2][3] = {{1,2,3},{4,5,6}};
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) printf("%d ", mat[i][j]);
        printf("\n");
    }
    return 0;
}

3. 小任務練習

#include <stdio.h>

int main(void) {
    int mat[2][3] = {{1,2,3},{4,5,6}};
    int row_sum[2] = {0};
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) row_sum[i] += mat[i][j];
    }
    printf("row0=%d row1=%d\n", row_sum[0], row_sum[1]);
    return 0;
}

反思提問

- 我是否確定索引都在合法範圍內？

- sizeof 計算元素數量有哪些限制？

- 二維陣列在記憶體中如何排列？

小型練習

- 找出一維陣列最大值與最小值。

- 計算二維陣列每欄總和。

7. 簡單指標

概念摘要

指標儲存位址，是 C 的核心能力之一。學習順序建議：取址 & -> 解參照 * -> 陣列與指標。

Step-by-step 操作

1. 最小可執行範例

#include <stdio.h>

int main(void) {
    int x = 42;
    int *p = &x;
    printf("x=%d, *p=%d\n", x, *p);
    return 0;
}

2. 小幅變化範例

#include <stdio.h>

int main(void) {
    int x = 10;
    int *p = &x;
    *p = 99;
    printf("x=%d\n", x);
    return 0;
}

3. 小任務練習

#include <stdio.h>

void fill_zero(int *arr, int n) {
    for (int i = 0; i < n; i++) arr[i] = 0;
}

int main(void) {
    int data[4] = {1, 2, 3, 4};
    fill_zero(data, 4);
    for (int i = 0; i < 4; i++) printf("%d ", data[i]);
    printf("\n");
    return 0;
}

反思提問

- p 與 *p 分別代表什麼？

- 為什麼 %p 要搭配 (void*)？

- 何時我應該用指標，何時只用變數就好？

小型練習

- 寫 swap(int *a, int *b) 交換兩數。

- 用指標走訪陣列並計算平均值。

8. 綜合實作：成績統計系統

A. 實作目標

1. 使用 2D 陣列存 5 位學生 3 科成績。

2. 用函式計算每人總分。

3. 使用判斷輸出 PASS/FAIL。

4. 用指標函式找最高總分。

B. 逐步構建（Learning-by-Doing）

1. Step 1：建立資料與輸出骨架

#include <stdio.h>
#define STUDENTS 5
#define SUBJECTS 3

int main(void) {
    int scores[STUDENTS][SUBJECTS] = {
        {80, 75, 90}, {60, 58, 70}, {95, 88, 92}, {40, 65, 55}, {78, 82, 79}
    };
    for (int i = 0; i < STUDENTS; i++) {
        printf("ID=%d -> %d %d %d\n", i, scores[i][0], scores[i][1], scores[i][2]);
    }
    return 0;
}

2. Step 2：加入總分函式

void calc_totals(int scores[STUDENTS][SUBJECTS], int totals[STUDENTS]) {
    for (int i = 0; i < STUDENTS; i++) {
        totals[i] = 0;
        for (int j = 0; j < SUBJECTS; j++) totals[i] += scores[i][j];
    }
}

3. Step 3：加入最高分查找與結果分類

void find_max_total(int totals[STUDENTS], int *max_idx, int *max_val) {
    *max_idx = 0;
    *max_val = totals[0];
    for (int i = 1; i < STUDENTS; i++) {
        if (totals[i] > *max_val) {
            *max_val = totals[i];
            *max_idx = i;
        }
    }
}

C. 完整程式碼範例

#include <stdio.h>

#define STUDENTS 5
#define SUBJECTS 3

void calc_totals(int scores[STUDENTS][SUBJECTS], int totals[STUDENTS]) {
    for (int i = 0; i < STUDENTS; i++) {
        totals[i] = 0;
        for (int j = 0; j < SUBJECTS; j++) totals[i] += scores[i][j];
    }
}

void find_max_total(int totals[STUDENTS], int *max_idx, int *max_val) {
    *max_idx = 0;
    *max_val = totals[0];
    for (int i = 1; i < STUDENTS; i++) {
        if (totals[i] > *max_val) {
            *max_val = totals[i];
            *max_idx = i;
        }
    }
}

int main(void) {
    int scores[STUDENTS][SUBJECTS] = {
        {80, 75, 90},
        {60, 58, 70},
        {95, 88, 92},
        {40, 65, 55},
        {78, 82, 79}
    };

    int totals[STUDENTS];
    calc_totals(scores, totals);

    printf("ID  Scores(3)        Total  Result\n");
    printf("-----------------------------------\n");
    for (int i = 0; i < STUDENTS; i++) {
        printf("%d   %3d %3d %3d    %3d    %s\n",
               i,
               scores[i][0], scores[i][1], scores[i][2],
               totals[i],
               (totals[i] >= 180) ? "PASS" : "FAIL");
    }

    int max_idx, max_val;
    find_max_total(totals, &max_idx, &max_val);
    printf("\nTop student: ID=%d, total=%d\n", max_idx, max_val);
    return 0;
}

D. 程式碼亮點解析

1. 二維陣列到一維摘要

for (int i = 0; i < STUDENTS; i++) {
    totals[i] = 0;
    for (int j = 0; j < SUBJECTS; j++) totals[i] += scores[i][j];
}

逐列累加，將每位學生的多科資料轉為單一總分。

2. 指標回傳多個結果

find_max_total(totals, &max_idx, &max_val);

用位址把「最高分學生 index」與「最高分值」同時帶回主流程。
